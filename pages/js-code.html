# JS代码段

/**
 * IPv6地址格式化，格式错误直接返回原数据
 */
export const ipv6Format = function(ip) {
  // 格式不确定
  if (!_.includes(ip, ':') ||
  (_.includes(ip, '::') && ip.split(':').length > 9) ||
  _.some(ip.split(':'), item => !_.includes(item, '.') && item.length > 4)) {
    return ip;
  }

  // ipv4in format
  const ipv4InFormat = _.chain(ip.split(':')).map(item => {
    if (_.includes(item, '.') && _.split(item, '.').length === 4) {
      const itemPart = _.chain(item).split('.').map(i => {
        const x = parseInt(i).toString(16);
        return x.length === 1 ? '0' + x : x;
      }).value();
      return `${itemPart[0]}${itemPart[1]}:${itemPart[2]}${itemPart[3]}`
    }
  }).join(':').value();

  // 格式不确定
  if (_.includes(ipv4InFormat, '.')) {
    return ip;
  }

  const ipPart = ipv4InFormat.split('::');

  // 格式不确定
  if (ipPart.length > 2) {
    return ip;
  }

  if (ipPart.length === 1) {
    // 格式不确定
    if (ipv4InFormat.split(':').length !== 8) {
      return ip;
    }

    // 补零
    return _.chain(ipv4InFormat.split(':'))
      .map(item => _.fill(new Array(4 - item.length), '0').join('') + item)
      .join(':')
      .toLower()
      .value();
  }

  const ipArray = _.fill(new Array(8), '0000');
  _.each(ipPart[0].split(':'), (item, index) => {
    ipArray[index] = _.fill(new Array(4 - item.length), '0').join('') + item;
  });
  _.each(ipPart[1].split(':'), (item, index) => {
    ipArray[8 - ipPart[1].split(':').length + index] = _.fill(new Array(4 - item.length), '0').join('') + item;
  });
  return ipArray.join(':').toLowerCase();
}
