# RAP2环境搭建

# Node

直接安装linux版本的node https://nodejs.org/dist/v10.16.0/node-v10.16.0-linux-x64.tar.gz

tar -xvf node-v10.16.0-linux-x64.tar.gz
cd node-v10.16.0-linux-x64.tar.gz
cp -r lib/node_modules /usr/local/lib
cp bin/node /usr/local/bin/

cd /usr/local/bin/
ln -s ../lib/node_modules/npm/bin/npm-cli.js npm
ln -s ../lib/node_modules/npm/bin/npx-cli.js npx

# mysql

# 下载mysql

https://downloads.mysql.com/archives/community

# 安装

主要安装4个包

mysql-community-common-5.7.28-1.el7.x86_64.rpm
mysql-community-libs-5.7.28-1.el7.x86_64.rpm
mysql-community-client-5.7.28-1.el7.x86_64.rpm
mysql-community-server-5.7.28-1.el7.x86_64.rpm

各个包有依赖关系需要顺序安装

rpm -ivh mysql-community-common-5.7.28-1.el7.x86_64.rpm
rpm -ivh mysql-community-libs-5.7.28-1.el7.x86_64.rpm
rpm -ivh mysql-community-client-5.7.28-1.el7.x86_64.rpm
rpm -ivh mysql-community-server-5.7.28-1.el7.x86_64.rpm

# 相关命令

systemctl stop mysqld.service
systemctl restart mysqld.service

# 修改密码

# 初始密码

MySQL第一次启动后会创建超级管理员账号root@localhost，初始密码储存在日志文件中

grep 'temporary password' /var/log/mysqld.log

# 修改默认密码

修改配置文件/etc/my.cnf，添加validate_password=OFF，保存并重启MySQL

mysql -uroot -p

mysql> ALTER USER 'root'@'localhost' IDENTIFIED BY '123456';
ERROR 1819 (HY000): Your password does not satisfy the current policy requirements

# 增加通过ip访问的mysql用户

mysql> grant all on *.* to admin@'%' identified by '123456' with grant option;
mysql> flush privileges;
# 查看用户
mysql> SELECT User, Host FROM mysql.user;

访问mysql -P 3306 -h 127.0.0.1 -u admin -p

# 修改mysql端口

vim /etc/my.cnf
port=3506

# 离线安装方式

在联网环境上通过npm i -g {node_module_name}的方式进行全局安装； 在联合环境的全局node_modules目录下找到组件，进行压缩，传送到离线环境； 拷贝组件到/usr/local/lib/node_modules/目录下，使用npm build -g {node_modules_name}进行安装

# Nginx

# 下载

http://nginx.org/download/nginx-1.15.0.tar.gz

# 依赖安装

yum -y install openssl openssl-devel
yum -y install pcre pcre-devel
yum -y install zlib zlib-devel

# 安装nginx

tar -zxvf nginx-1.15.0.tar.gz
cd nginx-1.15.0
./configure && make && make install

# 配置nginx

vim /usr/local/nginx/conf/nginx.conf

server {
  listen  80;
  server_name localhost;
  location / {
    root /home/rap2-dolores-master/build/;
    try_files $uri $uri @router;
  }

  location @router {
    rewrite ^.*$  /index.html last;
  }
}

# nginx常用相关命令

./nginx
./nginx -t
./nginx -s reload
tailf /usr/local/nginx/logs/access.log

# 主机重启后启动RAP2

cd /home/RAP2/rap2-delos/
npm run start:redis
npm start
cd /usr/local/nginx/sbin/
./nginx

# 设置开机启动

# 通过mp2 管理rap-server-delos 和 redis-server

pm2 list查看服务

pm2 save保存当前进程状态

pm2 startup [platform]生成开机自启动的命令

pm2 startup centos centos环境生成开机自启动命令

pm2 save保存

# Nginx开机启动

在系统服务目录中创建nginx.service文件

vim /lib/systemd/system/nginx.service

填写内容

[Unit]
Description=nginx
After=network.target

[Service]
Type=forking
ExecStart=/usr/local/nginx/sbin/nginx
ExecReload=/usr/local/nginx/sbin/nginx -s reload
ExecStop=/usr/local/nginx/sbin/nginx -s quit
PrivateTemp=true

[Install]
WantedBy=multi-user.target

systemctl daemon-reload生效配置

systemctl enable nginx.service nginx开机启动
